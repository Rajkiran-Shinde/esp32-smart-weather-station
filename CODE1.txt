// ---------- HELPER FUNCTIONS (Sensor Math) ----------
// (Unchanged functions: adcToVolts, dividerToVin, computeRs, getRs, getPPM)

float adcToVolts(int16_t raw) { return raw * (ADS_FS_VOLTAGE / ADS_COUNTS); }
float computeRs(float vin) { return (VCC_SENSOR - vin) * RL / vin; }

float getRs(uint8_t ch) {
  int16_t raw = ads.readADC_SingleEnded(ch);
  float vin = raw * (ADS_FS_VOLTAGE / ADS_COUNTS); // Direct MQ135 AOUT → ADS1115
  return (VCC_SENSOR - vin) * RL / vin;  // Standard MQ Rs formula
}

float getPPM(float Rs, float R0, float a, float b) {
  if (R0 <= 0 || Rs <= 0) return 0; // Prevent division by zero
  return a * pow((Rs / R0), b);
}

// ---------- CALIBRATION FUNCTION ----------
// (Unchanged)
float calibrateSensor(uint8_t channel, float clean_ratio, const char* label) {
  Serial.printf("Calibrating %s... Keep in clean air.\n", label);
  float sum = 0; int count = 0;
  for (int i = 0; i < 100; i++) {
    float Rs = getRs(channel);
    if (Rs > 0) { sum += Rs; count++; }
    delay(100); // Wait 100ms between reads (total 10 seconds)
  }
  float Rs_mean = sum / count;
  float R0 = Rs_mean / clean_ratio;
  Serial.printf("%s calibration complete. R0 = %.2f \u03A9\n", label, R0);
  return R0;
}

// ---------- SENSOR READING FUNCTIONS ----------
/**
 * Reads all MQ sensors and updates their PPM values (passed by reference).
 * MQ-7 removed.
 */
void mqReadings(float &ppm135, float &ppm4) {
  ppm135 = getPPM(getRs(CH_MQ135), R0_MQ135, MQ135_A, MQ135_B);
  // ppm7   = getPPM(getRs(CH_MQ7),   R0_MQ7,   MQ7_A,   MQ7_B); // REMOVED
  ppm4   = getPPM(getRs(CH_MQ4),   R0_MQ4,   MQ4_A,   MQ4_B);
}

/**
 * Reads BMP180 sensor (Pressure ONLY) and updates values (passed by reference).
 */
void bmpReadings(float &pressure, String &condition) {
  // temperature = bmp.readTemperature(); // REMOVED - Temp comes from DHT11
  pressure = bmp.readPressure() / 100.0;
  condition = getWeatherCondition(pressure);
}

/**
 * Reads DHT11 sensor and updates values (passed by reference).
 */
void dhtReadings(float &temperature, float &humidity) {
  humidity = dht.readHumidity();
  temperature = dht.readTemperature(); // Reads temperature as Celsius (the default)

  // Check if any reads failed and exit early (to prevent NaN issues)
  if (isnan(humidity) || isnan(temperature)) {
    Serial.println(F("Failed to read from DHT sensor! Keeping previous values."));
    // Keep last known good values if read fails
    // The initial values (0) are used if this fails before a good read.
  } else {
    // Only update globals if the read was successful
    temperature = temperature; // Assign to local variable
    humidity = humidity;       // Assign to local variable
  }
}


// ---------- CORE LOGIC FUNCTION (Non-Blocking) ----------

/**
 * Reads data from all sensors and updates the global variables.
 * DHT/BMP readings are always updated. MQ readings are updated only after calibration.
 */
void updateGlobalReadings() {
  
  float temp_check = current_temp; // Store current values
  float hum_check = current_humidity;
  
  // 1. DHT11 Reading (Temp, Humidity) - ALWAYS READ THESE
  dhtReadings(temp_check, hum_check);
  
  // Check if DHT read was successful before updating globals
  if (!isnan(temp_check) && !isnan(hum_check)) {
    current_temp = temp_check;
    current_humidity = hum_check;
  }
  
  // 2. BMP Reading (Pressure, Condition) - ALWAYS READ THESE
  bmpReadings(current_press, current_cond);

  // 3. Gas/PPM Readings - ONLY READ THESE IF CALIBRATED
  if (R0_MQ135 > 0 && R0_MQ4 > 0) { // MQ-7 check removed
    mqReadings(current_ppm135, current_ppm4); // MQ-7 removed
  } else {
    // Ensure MQ values are reset to 0 if not calibrated
    current_ppm135 = 0;
    current_ppm4 = 0;
  }

  // Print to Serial for debugging
  Serial.print("T: "); Serial.print(current_temp); 
  Serial.print("C, H: "); Serial.print(current_humidity); // Added Humidity
  Serial.print("%, P: "); Serial.print(current_press); 
  Serial.print("hPa, AQI: "); Serial.print(current_ppm135); Serial.println(" ppm");
}

// ---------- I2C Scanner (for Debugging) ----------
// (Unchanged)
void i2cScanner() {
  byte error, address;
  int nDevices;

  Serial.println(F("--- I2C Scanner Initiated ---"));
  nDevices = 0;
  for(address = 1; address < 127; address++ ) {
    Wire.beginTransmission(address);
    error = Wire.endTransmission();

    if (error == 0) {
      Serial.print(F("I2C device found at address 0x"));
      if (address<16) 
        Serial.print("0");
      Serial.println(address,HEX);
      nDevices++;
    }
  }
  if (nDevices == 0) {
    Serial.println(F("No I2C devices found. Check SDA (GPIO 21) and SCL (GPIO 22) wiring."));
  } else {
    Serial.print(nDevices);
    Serial.println(F(" device(s) found."));
  }
  Serial.println(F("--- Scan Complete ---"));
  Serial.println();
  delay(500); // Give time to read the scan
}

// ---------- SETUP ----------
void setup() {
  Serial.begin(115200);
  Wire.begin(SDA_PIN, SCL_PIN);
  Wire.setClock(100000);

  // Run the scanner to find devices before initializing libraries
  i2cScanner(); 
  
  if (!ads.begin(ADS1115_ADDRESS)) { // Using explicit 0x48 address
    Serial.println("❌ ADS1115 not found! Halting.");
    while (1);
  }
  ads.setGain(GAIN_ONE); // Set gain to +/- 4.096V

  if (!bmp.begin()) {
    Serial.println("❌ BMP180 not found! Halting.");
    while (1);
  }

  // Initialize DHT11
  dht.begin();
  Serial.println(F("DHT11 sensor initialized."));

  // Try 0x3C, then 0x3D for OLED
  bool oled_ok = false;
  if (display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) { 
    oled_ok = true;
    Serial.println(F("OLED found at 0x3C"));
  } else if (display.begin(SSD1306_SWITCHCAPVCC, 0x3D)) {
    oled_ok = true;
    Serial.println(F("OLED found at 0x3D"));
  }

  if (!oled_ok) {
    Serial.println("❌ OLED not found! Halting.");
    while (1);
  }

  Serial.println("✅ All devices initialized successfully!");
  display.clearDisplay();
  showProjectName();

  // Load Calibration Data from Non-Volatile Storage
  prefs.begin("calib", false);
  R0_MQ135 = prefs.getFloat("R0_135", 0);
  // R0_MQ7   = prefs.getFloat("R0_7", 0); // Removed
  R0_MQ4   = prefs.getFloat("R0_4", 0);
  prefs.end(); // Close preferences after reading

  if (R0_MQ135 > 0 && R0_MQ4 > 0) { // Modified check
    Serial.println("Loaded calibration data from memory.");
  } else {
    Serial.println("No calibration data found. Please calibrate.");
  }

  // --- PRE-LOAD INITIAL SENSOR DATA (New Stabilization Block) ---
  Serial.println(F("Delaying 2 seconds for sensor stabilization..."));
  delay(2000); // Wait for sensors (especially DHT) to stabilize
  updateGlobalReadings(); // Force initial read
  Serial.println(F("Initial data read complete. Starting server."));
  
  // --- Start Wi-Fi Connection (AP Mode) ---
  connectToWiFi();

  // --- Web Server Setup ---
  server.on("/", HTTP_GET, handleRoot); 
  server.on("/data", HTTP_GET, handleSensorData); 
  server.begin();
  Serial.println(F("HTTP server started."));

  // Prime the timers
  lastSensorReadTime = millis();
  lastOLEDUpdateTime = millis();
}

// ---------- MAIN LOOP (Non-Blocking) ----------
void loop() {
  
  // 1. CRITICAL: Handle Web Server Requests constantly!
  server.handleClient();

  // 2. Calibration Check (Non-blocking serial read)
  if (Serial.available()) {
    char cmd = Serial.read();
    if (cmd == 'c') {
      // Begin calibration process
      prefs.begin("calib", false); // Open preferences in R/W mode
      R0_MQ135 = calibrateSensor(CH_MQ135, RATIO_CLEAN_MQ135, "MQ135");
      // R0_MQ7   = calibrateSensor(CH_MQ7,   RATIO_CLEAN_MQ7,   "MQ7"); // Removed
      R0_MQ4   = calibrateSensor(CH_MQ4,   RATIO_CLEAN_MQ4,   "MQ4");

      // Save new R0 values to memory
      prefs.putFloat("R0_135", R0_MQ135);
      // prefs.putFloat("R0_7",   R0_MQ7); // Removed
      prefs.putFloat("R0_4",   R0_MQ4);
      prefs.end();
      Serial.println("✅ Calibration data saved.\n");
    
    } else if (cmd == 'r') {
      // Clear calibration data
      prefs.begin("calib", false);
      prefs.clear();
      prefs.end();
      Serial.println("❌ Calibration data cleared. Restarting...");
      delay(200);
      ESP.restart();
    }
  }

  // 3. Sensor Reading Interval (Non-blocking: Reads data every 5 seconds)
  if (millis() - lastSensorReadTime >= sensorInterval) {
    lastSensorReadTime = millis();
    updateGlobalReadings(); // Reads all sensors, updates global data, and prints to serial
  }

  // 4. OLED Display Rotation (Non-blocking: Rotates screen every 3 seconds)
  if (millis() - lastOLEDUpdateTime >= oledInterval) {
    lastOLEDUpdateTime = millis();

    if (R0_MQ135 > 0 && R0_MQ4 > 0) { // Modified check
      // Cycle between the two data screens
      if (currentOLED_Screen == 1) {
        showGasScreen(current_ppm135, current_ppm4); // Modified call
        currentOLED_Screen = 2;
      } else {
        showWeatherScreen(current_temp, current_humidity, current_press, current_cond); // Modified call
        currentOLED_Screen = 1;
      }
    } else {
      // Show Calibration screen if R0 is 0
      display.clearDisplay();
      display.setCursor(0, 20);
      display.setTextSize(1);
      display.print("Type 'c' in Serial");
      display.setCursor(0, 32);
      display.print("to Calibrate Sensors");
      display.display();
    }
  }
}
